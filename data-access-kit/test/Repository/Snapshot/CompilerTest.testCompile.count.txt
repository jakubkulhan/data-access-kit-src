<?php declare(strict_types=1);

namespace DataAccessKit\Repository\Fixture;

use DataAccessKit\Attribute\Column;
use DataAccessKit\PersistenceInterface;
use DataAccessKit\Repository\Exception\MultipleObjectsFoundException;
use DataAccessKit\Repository\Exception\NotFoundException;

final class CountRepository implements CountRepositoryInterface
{
	private object $countByTitleArguments;

	public function __construct(
		private readonly PersistenceInterface $persistence,
	)
	{
		$this->countByTitleArguments = new class {
			#[Column(name: "title")]
			public string $title;
		};
	}

	public function countByTitle(
		string $title,
	): int
	{
		$arguments = clone $this->countByTitleArguments;
		$arguments->title = $title;
		$arguments = $this->persistence->toRow($arguments);

		$result = $this->persistence->query(int::class, 'SELECT COUNT(*) FROM foos t WHERE t.title = ?', [$arguments['title']]);

		$objects = iterator_to_array($result);
		if (count($objects) === 0) {
			throw new NotFoundException('int');
		} else if (count($objects) > 1) {
			throw new MultipleObjectsFoundException('int');
		}
		return $objects[0];
	}
}
